#include <stdio.h>

 int main (int argc, char *argv[]) {
  int i=0; char card[90];

  fprintf(stdout,"\n");
  fgets(card, 80, stdin);
  while (!feof(stdin)) {
   i++;
   fprintf(stdout, "%5d\t%s", i, card);
   fgets(card, 80, stdin);
   }
  fprintf(stdout,"\n\n");
 }
